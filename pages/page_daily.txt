def safe_int(val):
    """ìˆ«ì ì»¬ëŸ¼ ì•ˆì „ ë³€í™˜ìš©"""
    try:
        if val is None:
            return 0
        s = str(val).strip()
        if s == "":
            return 0
        return int(float(s.replace(",", "")))
    except (TypeError, ValueError):
        return 0


def _pack_memo(user_memo: str, income_type: str, exp1_type: str, exp2_type: str) -> str:
    """
    âœ… ì‹œíŠ¸ ì»¬ëŸ¼ì„ ëŠ˜ë¦¬ì§€ ì•Šê³ ,
    ë¯¸ìˆ˜(ìˆ˜ìµ ì œì™¸) / ì§€ì¶œ ìˆ˜ë‹¨ ì •ë³´ë¥¼ memo ì•ˆì— íƒœê·¸ë¡œ ì €ì¥.
    """
    user_memo = (user_memo or "").strip()
    tag = f"[KID]inc={income_type or ''};e1={exp1_type or ''};e2={exp2_type or ''}[/KID]"
    if user_memo:
        return f"{tag} {user_memo}"
    return tag


def _unpack_memo(memo: str) -> tuple[dict, str]:
    """
    memoì—ì„œ [KID]...[/KID] íƒœê·¸ë¥¼ ì½ì–´ metaì™€ ì‚¬ìš©ì ë¹„ê³ ë¥¼ ë¶„ë¦¬.
    """
    memo = memo or ""
    meta = {"inc": "", "e1": "", "e2": ""}
    user = memo

    try:
        start = memo.find("[KID]")
        end = memo.find("[/KID]")
        if start != -1 and end != -1 and end > start:
            inner = memo[start + 5:end]
            for part in inner.split(";"):
                if "=" in part:
                    k, v = part.split("=", 1)
                    k = k.strip()
                    v = v.strip()
                    if k in meta:
                        meta[k] = v
            user = (memo[end + 6:] or "").strip()
    except Exception:
        pass

    return meta, user

def upsert_daily_records(records: list[dict]) -> bool:
    """
    âœ… 'ì¼ì¼ê²°ì‚°' ì‹œíŠ¸ì— id ê¸°ì¤€ìœ¼ë¡œ í–‰ì„ ì¶”ê°€/ìˆ˜ì •(upsert)í•©ë‹ˆë‹¤.
    - ì „ì²´ ë®ì–´ì“°ê¸° ì•„ë‹˜
    """
    header = [
        "id",
        "date",
        "time",
        "category",
        "name",
        "task",
        "income_cash",
        "income_etc",
        "exp_cash",
        "cash_out",
        "exp_etc",
        "memo",
    ]

    try:
        client = get_gspread_client()
        ws = get_worksheet(client, DAILY_SUMMARY_SHEET_NAME)

        values = ws.get_all_values()
        if not values:
            # ì‹œíŠ¸ê°€ ë¹„ì–´ìˆìœ¼ë©´: í—¤ë” + records ì „ì²´
            rows = [header]
            for rec in records:
                rows.append([str(rec.get(h, "")) for h in header])
            ws.update(rows)
            return True

        sheet_header = values[0]
        if "id" not in sheet_header:
            # í—¤ë”ê°€ ê¹¨ì§„ ê²½ìš°: í—¤ë”ë¶€í„° ì •ìƒí™”(ìµœì†Œ ì•ˆì „ì¥ì¹˜)
            ws.update([header])
            sheet_header = header

        id_col = sheet_header.index("id")

        # id -> ì‹œíŠ¸ rowë²ˆí˜¸(2ë¶€í„° ì‹œì‘)
        existing = {}
        for r_i, row in enumerate(values[1:], start=2):
            rid = row[id_col].strip() if len(row) > id_col else ""
            if rid:
                existing[rid] = r_i

        # í—¤ë” ê¸¸ì´ì— ë§ì¶° A~ ëì—´ ë²”ìœ„ ê³„ì‚°
        def _col_letter(n: int) -> str:
            # 1->A, 26->Z, 27->AA
            s = ""
            while n:
                n, r = divmod(n - 1, 26)
                s = chr(65 + r) + s
            return s

        last_col = _col_letter(len(header))

        for rec in records:
            rid = str(rec.get("id", "")).strip()
            if not rid:
                rid = str(uuid.uuid4())
                rec["id"] = rid

            row_vals = [str(rec.get(h, "")) for h in header]

            if rid in existing:
                row_no = existing[rid]
                ws.update(f"A{row_no}:{last_col}{row_no}", [row_vals])
            else:
                ws.append_row(row_vals)

        return True

    except Exception as e:
        st.error(f"âŒ ì¼ì¼ê²°ì‚° ì €ì¥ ì‹¤íŒ¨: {e}")
        return False

def delete_daily_record_by_id(record_id: str) -> bool:
    """âœ… 'ì¼ì¼ê²°ì‚°' ì‹œíŠ¸ì—ì„œ idê°€ record_idì¸ í–‰ 1ê°œë§Œ ì‚­ì œ"""
    try:
        client = get_gspread_client()
        ws = get_worksheet(client, DAILY_SUMMARY_SHEET_NAME)
        values = ws.get_all_values()
        if not values:
            return False

        header = values[0]
        if "id" not in header:

        id_col = header.index("id")

        for row_no, row in enumerate(values[1:], start=2):
            rid = row[id_col].strip() if len(row) > id_col else ""

                ws.delete_rows(row_no)

        return False

    except Exception as e:
        st.error(f"âŒ ì‚­ì œ ì‹¤íŒ¨: {e}")
        return False

def _norm(x) -> str:
    return str(x or "").strip()


def _ensure_active_tasks_header(ws, header_needed: list[str]) -> list[str]:
    """âœ… í—¤ë”ëŠ” 'ë®ì–´ì“°ê¸°/ì¬ì •ë ¬' ê¸ˆì§€. í•„ìš”í•œ ì»¬ëŸ¼ë§Œ **ëì— ì¶”ê°€**í•œë‹¤."""
    values = ws.get_all_values()
    if not values:
        # ì™„ì „ ë¹ˆ ì‹œíŠ¸ë¼ë©´ í—¤ë”ë§Œ ë¨¼ì € ìƒì„±
        ws.update(f"A1:{_col_letter(len(header_needed))}1", [header_needed])
        return header_needed

    header = values[0]
    missing = [h for h in header_needed if h not in header]
    if missing:
        new_header = header + missing
        ws.update(f"A1:{_col_letter(len(new_header))}1", [new_header])
        return new_header

    return header


def _repair_active_tasks_shift_if_needed(ws, header: list[str]) -> None:
    """
    âœ… ê³¼ê±° ë°ì´í„°ê°€ 'í—¤ë” ê°•ì œ êµì²´'ë¡œ ì¸í•´ ë°€ë¦° ê²½ìš° ë³µêµ¬.
    - ì¦ìƒ: cash ì»¬ëŸ¼ì— TRUE/FALSE, planned_expense ê°€ ë¹„ì–´ìˆê³ , transfer ì— ìˆ«ìê°€ ë“¤ì–´ìˆìŒ
    - ë³µêµ¬: planned_expense <- transfer, processed <- cash, processed_timestamp <- card
            transfer/cash/card -> 0
    """
    need_cols = ["transfer", "cash", "card", "planned_expense", "processed", "processed_timestamp"]
    if any(c not in header for c in need_cols):
        return

    idx = {c: header.index(c) for c in need_cols}
    # transfer..processed_timestamp ê°€ ì—°ì†ì´ë©´ í•œ ë²ˆì— ì—…ë°ì´íŠ¸ ê°€ëŠ¥
    start_i = min(idx.values())
    end_i = max(idx.values())

    values = ws.get_all_values()
    if len(values) <= 1:
        return

    ranges = []
    payloads = []

    for row_no, row in enumerate(values[1:], start=2):
        cash_v = row[idx["cash"]] if idx["cash"] < len(row) else ""
        proc_v = row[idx["processed"]] if idx["processed"] < len(row) else ""
        tr_v = row[idx["transfer"]] if idx["transfer"] < len(row) else ""
        card_v = row[idx["card"]] if idx["card"] < len(row) else ""

        cash_s = str(cash_v).strip().upper()
        # 'ë°€ë¦¼' íœ´ë¦¬ìŠ¤í‹±
        if cash_s in ("TRUE", "FALSE") and str(proc_v).strip() == "" and str(tr_v).strip().isdigit():
            new_tr = "0"
            new_cash = "0"
            new_card = "0"
            new_planned = str(tr_v).strip()
            new_proc = cash_s
            new_ts = str(card_v).strip()

            # í•„ìš”í•œ ì»¬ëŸ¼ë“¤ì„ í•œ ë²ˆì— ì—…ë°ì´íŠ¸(ì—°ì† ë²”ìœ„)
            row_out = []
            for col_i in range(start_i, end_i + 1):
                # ê¸°ë³¸ì€ ì›ë˜ ê°’ ìœ ì§€
                v = row[col_i] if col_i < len(row) else ""
                row_out.append(v)

            # ë®ì–´ì“¸ ìœ„ì¹˜
            row_out[idx["transfer"] - start_i] = new_tr
            row_out[idx["cash"] - start_i] = new_cash
            row_out[idx["card"] - start_i] = new_card
            row_out[idx["planned_expense"] - start_i] = new_planned
            row_out[idx["processed"] - start_i] = new_proc
            row_out[idx["processed_timestamp"] - start_i] = new_ts

            a1 = f"{_col_letter(start_i+1)}{row_no}:{_col_letter(end_i+1)}{row_no}"
            ranges.append(a1)
            payloads.append([row_out])

    if ranges:
        # ì—¬ëŸ¬ í–‰ì„ ê°œë³„ ë²”ìœ„ë¡œ ì—…ë°ì´íŠ¸ (ì•ˆì „: í•„ìš”í•œ ì…€ë§Œ)
        batch = [{"range": r, "values": v} for r, v in zip(ranges, payloads)]
        ws.batch_update(batch)

    header = values[0]
    # í•„ìš”í•œ ì»¬ëŸ¼ì´ í•˜ë‚˜ë¼ë„ ì—†ìœ¼ë©´, í—¤ë”ë¥¼ V2ë¡œ ê°•ì œ í™•ì¥ (ë°ì´í„°ëŠ” ìœ ì§€)
    missing = [h for h in header_needed if h not in header]
    if missing:
        ws.update(f"A1:{_col_letter(len(header_needed))}1", [header_needed])
        return header_needed

    return header


def _col_letter(n: int) -> str:
    s = ""
    while n:
        n, r = divmod(n - 1, 26)
        s = chr(65 + r) + s
    return s


def upsert_active_task_records(records: list[dict]) -> bool:
    """âœ… ì§„í–‰ì—…ë¬´ ì‹œíŠ¸ì— id ê¸°ì¤€ upsert (ì „ì²´ ë®ì–´ì“°ê¸° ê¸ˆì§€)"""
    try:
        if not records:
            return True

        client = get_gspread_client()
        ws = get_worksheet(client, ACTIVE_TASKS_SHEET_NAME)

        header = _ensure_active_tasks_header(ws, ACTIVE_TASKS_HEADER_V2)
        values = ws.get_all_values()
        header = values[0] if values else header

        if "id" not in header:
            return False

        id_col = header.index("id")
        existing = {}
        for row_no, row in enumerate(values[1:], start=2):
            rid = row[id_col].strip() if len(row) > id_col else ""
            if rid:
                existing[rid] = row_no

        last_col = _col_letter(len(header))

        for rec in records:
            rid = _norm(rec.get("id"))
            if not rid:
                rid = str(uuid.uuid4())
                rec["id"] = rid

            row_vals = [str(rec.get(h, "")) for h in header]

            if rid in existing:
                row_no = existing[rid]
                ws.update(f"A{row_no}:{last_col}{row_no}", [row_vals])
            else:
                ws.append_row(row_vals)

        return True

    except Exception as e:
        st.error(f"âŒ ì§„í–‰ì—…ë¬´ ë°˜ì˜ ì‹¤íŒ¨: {e}")
        return False




def apply_daily_to_active_tasks(
    *,
    date_str: str,
    category: str,
    name: str,
    work: str,
    memo_user: str,
    income_type: str,
    income_amt: int,
    exp1_type: str,
    exp1_amt: int,
    exp2_type: str,
    exp2_amt: int,
) -> bool:
    """ì¼ì¼ê²°ì‚° 1ê±´ì„ ì§„í–‰ì—…ë¬´ì— ëˆ„ì  ë°˜ì˜í•œë‹¤."""
    try:
        # 1) ì¦ê°ì•¡ ê³„ì‚°
        add_transfer = 0
        add_cash = 0
        add_card = 0
        add_stamp = 0
        add_receivable = 0

        def _add_exp(t: str, a: int):
            nonlocal add_transfer, add_cash, add_card, add_stamp
            if a <= 0:
                return
            if t == "ì´ì²´":
                add_transfer += a
            elif t == "í˜„ê¸ˆ":
                add_cash += a
            elif t == "ì¹´ë“œ":
                add_card += a
            elif t == "ì¸ì§€":
                add_stamp += a

        _add_exp(exp1_type, exp1_amt)
        _add_exp(exp2_type, exp2_amt)

        if income_type == "ë¯¸ìˆ˜" and income_amt > 0:
            add_receivable += income_amt

        # 2) ê¸°ì¡´ ì§„í–‰ì—…ë¬´ ë¡œë“œ (âœ… ìºì‹œ ìš°íšŒ: gspreadë¡œ ì§ì ‘ ì½ê¸°)
        client = get_gspread_client()
        ws = get_worksheet(client, ACTIVE_TASKS_SHEET_NAME)
        _ = _ensure_active_tasks_header(ws, ACTIVE_TASKS_HEADER_V2)
        _repair_active_tasks_shift_if_needed(ws, ws.get_all_values()[0])

        values = ws.get_all_values() or []

        tasks: list[dict] = []
        if values and len(values) > 1:
            header = values[0]
            for row in values[1:]:
                rec = {col: (row[i] if i < len(row) else "") for i, col in enumerate(header)}
                tasks.append(rec)

        # 3) ë§¤ì¹­: (category, date, name, work)
        target = None
        for t in tasks:
            if (
                _norm(t.get("category")) == _norm(category)
                and _norm(t.get("date")) == _norm(date_str)
                and _norm(t.get("name")) == _norm(name)
                and _norm(t.get("work")) == _norm(work)
            ):
                target = dict(t)
                break

        if target is None:
            target = {
                "id": str(uuid.uuid4()),
                "category": category,
                "date": date_str,
                "name": name,
                "work": work,
                "source_original": "daily",
                "details": memo_user or "",
                "transfer": 0,
                "cash": 0,
                "card": 0,
                "stamp": 0,
                "receivable": 0,
                "planned_expense": 0,
                "processed": "FALSE",
                "processed_timestamp": "",
            }

        # 4) ëˆ„ì 
        target["transfer"] = safe_int(target.get("transfer")) + add_transfer
        target["cash"] = safe_int(target.get("cash")) + add_cash
        target["card"] = safe_int(target.get("card")) + add_card
        target["stamp"] = safe_int(target.get("stamp")) + add_stamp
        target["receivable"] = safe_int(target.get("receivable")) + add_receivable

        target["planned_expense"] = (
            safe_int(target.get("transfer"))
            + safe_int(target.get("cash"))
            + safe_int(target.get("card"))
            + safe_int(target.get("stamp"))
        )

        # 5) id ê¸°ì¤€ upsert
        return upsert_active_task_records([target])

    except Exception as e:
        st.error(f"âŒ ì§„í–‰ì—…ë¬´ ë°˜ì˜ ì˜¤ë¥˜: {e}")
        return False

def load_daily():
    """
    âœ… 'ì¼ì¼ê²°ì‚°' ì‹œíŠ¸ í—¤ë”(ì‚¬ìš©ì ì œê³µ):

            "time": str(r.get("time", "")),
            "category": str(r.get("category", "")),
            "name": str(r.get("name", "")),
            "task": str(r.get("task", "")),
            "income_cash": safe_int(r.get("income_cash")),
            "income_etc": safe_int(r.get("income_etc")),
            "exp_cash": safe_int(r.get("exp_cash")),
            "cash_out": safe_int(r.get("cash_out")),
            "exp_etc": safe_int(r.get("exp_etc")),
            "memo": str(r.get("memo", "")),
        }
        processed_records.append(entry)
    return processed_records

def load_balance():
    records = read_data_from_sheet(DAILY_BALANCE_SHEET_NAME, default_if_empty=[])
    balance = {"cash": 0, "profit": 0}
    if not records:
        return balance

    for record in records:
        key = record.get("key")
        value_str = str(record.get("value", "0"))
        if key in balance:
            try:
                balance[key] = int(value_str) if value_str and value_str.strip() else 0
            except ValueError:
                st.warning(f"ëˆ„ì ìš”ì•½ ë°ì´í„° '{key}' ê°’ '{value_str}' ìˆ«ì ë³€í™˜ ì‹¤íŒ¨ â†’ 0ìœ¼ë¡œ ì²˜ë¦¬í•©ë‹ˆë‹¤.")
                balance[key] = 0
    return balance
